📊 Required Firestore Security Rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /user-visits/{document} {
      allow read, write: if true;
    }
    match /user-downloads/{document} {
      allow read, write: if true;
    }
    match /admin-zues-oun/{document} {
      allow read: if true;
      allow write: if true;
    }
  }
}
